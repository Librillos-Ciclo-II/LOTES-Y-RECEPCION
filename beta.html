// ── Inicializa/completa solapa CONFIG (llamar sin arg para ejecución manual) ── function initConfig(sheet) { const ss2 = SpreadsheetApp.openById(SHEET_ID); if (!sheet) { sheet = ss2.getSheetByName('CONFIG') || ss2.insertSheet('CONFIG'); } if (!sheet.getRange('A1').getValue()) { sheet.getRange('A1').setValue('PASS_SUPERVISOR'); sheet.getRange('B1').setValue('1234'); sheet.getRange('A2').setValue('PASS_ADMIN'); sheet.getRange('B2').setValue('admin1234'); sheet.getRange('A3').setValue('--- Cambiar contraseñas en col B ---'); sheet.getRange('A3').setFontColor('#999999'); } if (!sheet.getRange('A4').getValue()) { sheet.getRange('A4').setValue('FCL_NUM_INICIO'); sheet.getRange('B4').setValue(11); sheet.getRange('A5').setValue('FCL_FECHA_INICIO'); sheet.getRange('B5').setValue('2026-04-13'); sheet.getRange('A6').setValue('FCL_KGS_INICIO'); sheet.getRange('B6').setValue(13880); sheet.getRange('A7').setValue('FCL_CAPACIDAD_KG'); sheet.getRange('B7').setValue(28000); sheet.getRange('A1:A7').setFontWeight('bold'); } Logger.log('CONFIG OK: FCL_NUM=' + sheet.getRange('B4').getValue() + ' | FECHA=' + sheet.getRange('B5').getValue() + ' | KGS=' + sheet.getRange('B6').getValue() + ' | CAP=' + sheet.getRange('B7').getValue()); } function doGet(e) { try { const action = e.parameter.action; const ss = SpreadsheetApp.openById(SHEET_ID); // ── GET ────────────────────────────────────────────────────────────────── if (action === 'get') { const sheet = ss.getSheetByName(e.parameter.sheet); if (!sheet) return resp({ok:false, error:'Sheet not found: '+e.parameter.sheet}); return resp({ok:true, values: sheet.getDataRange().getValues()}); } // ── GET_ALL — una sola llamada trae todo (elimina cold start duplicado) ── if (action === 'get_all') { const shRec = ss.getSheetByName('RECEPCIONES'); const shLot = ss.getSheetByName('LOTES'); let shCfg = ss.getSheetByName('CONFIG'); if (!shCfg) shCfg = ss.insertSheet('CONFIG'); initConfig(shCfg); const recepciones = shRec ? shRec.getDataRange().getValues() : []; const lotes = shLot ? shLot.getDataRange().getValues() : []; const config = { passSupervisor: String(shCfg.getRange('B1').getValue() || '1234'), passAdmin: String(shCfg.getRange('B2').getValue() || 'admin1234'), fclNumInicio: Number(shCfg.getRange('B4').getValue()) || 11, fclFechaInicio: String(shCfg.getRange('B5').getValue() || '2026-04-13'), fclKgsInicio: Number(shCfg.getRange('B6').getValue()) || 13880, fclCapacidadKg: Number(shCfg.getRange('B7').getValue()) || 28000, }; return resp({ok:true, recepciones, lotes, config}); } // ── POST (append row) ──────────────────────────────────────────────────── if (action === 'post') { const data = JSON.parse(decodeURIComponent(e.parameter.data)); const sheet = ss.getSheetByName(data.sheet); if (!sheet) return resp({ok:false, error:'Sheet not found: '+data.sheet}); sheet.appendRow(data.row); return resp({ok:true}); } // ── UPDATE RECEPCION (modificar fila por ID) ───────────────────────────── if (action === 'update_recepcion') { const data = JSON.parse(decodeURIComponent(e.parameter.data)); const sheet = ss.getSheetByName('RECEPCIONES'); const values = sheet.getDataRange().getValues(); let rowIdx = -1; for (let i = 1; i < values.length; i++) { if (values[i][0] === data.id) { rowIdx = i + 1; break; } } if (rowIdx === -1) return resp({ok:false, error:'Recepcion no encontrada: '+data.id}); // Cols: D=Unidades(4), E=Tambores(5), F=Comentarios(6) sheet.getRange(rowIdx, 4).setValue(data.unidades); sheet.getRange(rowIdx, 5).setValue(data.tambores || ''); sheet.getRange(rowIdx, 6).setValue(data.comentarios || ''); return resp({ok:true}); } // ── UPDATE TOTAL_LOTE en LOTES (cuando se modifica recepción) ──────────── if (action === 'update_total_lote') { const data = JSON.parse(decodeURIComponent(e.parameter.data)); const sheet = ss.getSheetByName('LOTES'); const values = sheet.getDataRange().getValues(); let updated = 0; for (let i = 1; i < values.length; i++) { if (values[i][0] === data.id_recepcion) { // Col I = Total_Lote (índice 8, col 9) sheet.getRange(i + 1, 9).setValue(data.nuevo_total); updated++; } } return resp({ok:true, updated}); } // ── GET CONFIG ──────────────────────────────────────────────────────────────── if (action === 'get_config') { let sheet = ss.getSheetByName('CONFIG'); if (!sheet) sheet = ss.insertSheet('CONFIG'); initConfig(sheet); const passSupervisor = String(sheet.getRange('B1').getValue()); const passAdmin = String(sheet.getRange('B2').getValue()); const fclNumInicio = Number(sheet.getRange('B4').getValue()) || 11; const fclFechaInicio = String(sheet.getRange('B5').getValue() || '2026-04-13'); const fclKgsInicio = Number(sheet.getRange('B6').getValue()) || 13880; const fclCapacidadKg = Number(sheet.getRange('B7').getValue()) || 28000; return resp({ok:true, passSupervisor, passAdmin, pass: passSupervisor, fclNumInicio, fclFechaInicio, fclKgsInicio, fclCapacidadKg}); } return resp({ok:false, error:'Unknown action: '+action}); } catch(err) { return resp({ok:false, error:err.toString()}); } } function resp(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
